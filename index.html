Supabase SQL Setup

Run this in your Supabase → SQL Editor once. All column names are lowercase, no underscore confusion.

-- ═══════════════════════════════════════════
-- OVEN X INVENTORY — SUPABASE SQL SETUP
-- Run this ONCE in Supabase SQL Editor
-- All columns lowercase, no ambiguity
-- ═══════════════════════════════════════════

-- 1. INGREDIENTS (master list)
create table if not exists ingredients (
  id           text primary key,
  name         text not null,
  category     text not null default 'general',
  unit         text not null default 'pcs',
  currentstock numeric not null default 0,
  openingstock numeric not null default 0,
  openingdate  date,
  minlevel     numeric not null default 0,
  costperunit  numeric not null default 0,
  supplier     text default '',
  location     text default '',
  updatedat    timestamptz default now()
);

-- 2. GRN HEADER
create table if not exists grn (
  id           text primary key,
  bizday       date not null,
  supplier     text not null,
  invoiceno    text default '',
  grandtotal   numeric default 0,
  notes        text default '',
  createdat    timestamptz default now()
);

-- 3. GRN ITEMS
create table if not exists grn_items (
  id            text primary key,
  grn_id        text references grn(id) on delete cascade,
  ingredient_id text references ingredients(id),
  itemname      text not null,
  qty           numeric not null,
  unit          text not null,
  unitcost      numeric not null,
  linetotal     numeric not null
);

-- 4. CLOSING HEADER
create table if not exists closing (
  id           text primary key,
  bizday       date not null unique,
  closedby     text not null,
  totalcost    numeric default 0,
  createdat    timestamptz default now()
);

-- 5. CLOSING ITEMS
create table if not exists closing_items (
  id            text primary key,
  closing_id    text references closing(id) on delete cascade,
  ingredient_id text references ingredients(id),
  itemname      text not null,
  unit          text not null,
  opening       numeric default 0,
  grntoday      numeric default 0,
  available     numeric default 0,
  physical      numeric default 0,
  consumed      numeric default 0,
  costperunit   numeric default 0,
  consumptioncost numeric default 0
);

-- Enable Realtime (optional)
alter publication supabase_realtime add table ingredients;
alter publication supabase_realtime add table grn;
alter publication supabase_realtime add table closing;
